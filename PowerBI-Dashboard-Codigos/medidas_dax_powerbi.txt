-- 1. Receita Total
Receita Total = SUM('Base Vendas'[Faturamento na Venda])

-- 2. Produtos Vendidos
Produtos Vendidos = SUM('Base Vendas'[Qtd Devolvida])
-- ou
-- Produtos Vendidos = SUM('Base Vendas'[Quantidade Vendida])

-- 3. Ticket Médio
Ticket Médio = 
DIVIDE(
    [Receita Total],
    [Produtos Vendidos],
    0
)

-- 4. Quantidade de Clientes
Quantidade de Clientes = DISTINCTCOUNT('Base Vendas'[Id Cliente])

-- 5. Faturamento por Mês
Faturamento por Mês = 
CALCULATE(
    [Receita Total],
    ALLEXCEPT('Base Vendas', 'Base Vendas'[Data Venda])
)
